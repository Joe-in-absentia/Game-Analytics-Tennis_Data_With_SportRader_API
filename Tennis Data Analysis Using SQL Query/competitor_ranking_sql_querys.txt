--Get all competitors with their rank and points.

SELECT DISTINCT c.name,cr.rank,cr.points
FROM competitors_table c 
JOIN competitor_ranking_table cr ON c.competitor_id = cr.competitor_id


--Find competitors ranked in the top 5.

SELECT DISTINCT c.competitor_id,c.name,c.country,c.abbreviation,cr.rank,cr.points
FROM competitors_table c
JOIN competitor_ranking_table cr ON c.competitor_id = cr.competitor_id
WHERE cr.rank <= 5
ORDER BY cr.rank


--List competitors with no rank movement (stable rank).

SELECT c.competitor_id,c.name,c.country,c.abbreviation,cr.rank,cr.points,cr.movement
FROM competitors_table c
JOIN competitor_ranking_table cr ON c.competitor_id = cr.competitor_id
WHERE cr.movement = 0


--Get the total points of competitors from a specific country (e.g., Croatia).

SELECT c.country, SUM(cr.points) AS total_points
FROM competitors_table c
JOIN competitor_ranking_table cr ON c.competitor_id = cr.competitor_id
WHERE c.country = 'Croatia'
GROUP BY c.country


--Count the number of competitors per country.

SELECT c.country, COUNT(c.competitor_id) AS competitor_count
FROM competitors_table c
JOIN competitor_ranking_table cr ON c.competitor_id = cr.competitor_id
GROUP BY c.country
ORDER BY competitor_count DESC;
